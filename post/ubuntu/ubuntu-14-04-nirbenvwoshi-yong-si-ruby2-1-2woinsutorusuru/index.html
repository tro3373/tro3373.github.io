Ubuntu 14.04 にrbenvを使用し、ruby2.1.2をインストールする

2014-09-18
#ubuntu  #ubuntu14.04  #rbenv  #Ruby 

install packages

sudo add-apt-repository ppa:chris-lea/node.js
sudo apt-get update
sudo apt-get install build-essential git curl zlib1g-dev libssl-dev libreadline-dev libyaml-dev libxml2-dev libxslt-dev sqlite3 libsqlite3-dev nodejs tk ruby-dev

rbenv install

git clone https://github.com/sstephenson/rbenv.git ~/.rbenv
git clone https://github.com/sstephenson/ruby-build.git ~/.rbenv/plugins/ruby-build
echo 'export PATH="$HOME/.rbenv/bin:$PATH"' >> ~/.bashrc
echo 'eval "$(rbenv init -)"' >> ~/.bashrc
. .bashrc
gem install rbenv-rehash
gem install bundler

ruby install

rbenv install -list
rbenv install 2.1.2
rbenv versions
rbenv global 2.1.2

rails

gem install rails
rbenv rehash
rails -v
